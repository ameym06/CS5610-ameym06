CNN World News Using Google Feed API


In this experiment we use the RSS feed using Google-Feed API and display dynamic content from the "CNN World" page

Code Documentation

function loadFeed(result) {
if (!result.error) {
var container = document.getElementById("region");
var html = "";
for (var i = 0; i < result.feed.entries.length; i++) {
var entry = result.feed.entries[i];
var div = document.createElement("div");
var title = entry.title;
var description = entry.content
var link = entry.link;
var pubDate = entry.publishedDate;
html += "";
html += title;
html += "";
html += "

" + pubDate + "

";
html += "

" + description + "

";
html += "
";
div.innerHTML = html
container.appendChild(div);
} } }
