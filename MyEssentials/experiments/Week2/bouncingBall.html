Bouncy Ball


Description

This canvas contains a bouncing ball. The ball changes its directions whenever it hits the canvas edges. This experiment is implemented using Canvas Graphics.

Code Documentation

unction animate() {
var canvas = $('#bounce_canvas')[0];
var ctx = canvas.getContext("2d");
ctx.clearRect(0, 0, width * 2, height * 2);
$('#canvas-height').html(height);
$('#canvas-width').html(width);
ctx.beginPath();
ctx.fillStyle = "black";
ctx.arc(ballwidth, ballheight, 10, 0, Math.PI * 2, true);
ctx.closePath();
ctx.fill();
if (ballwidth < 0 || ballwidth > width) x_step = -x_step;
if (ballheight < 0 || ballheight > height) y_step = -y_step;
ballwidth += x_step;
ballheight += y_step;
}
